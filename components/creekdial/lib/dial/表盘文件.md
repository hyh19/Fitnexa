typedef struct plate_management_file_head
{
    char file_name[30];//表盘名字

    uint16_t file_count;        //文件个数
}plate_management_file_head_t;
//单个文件信息
typedef struct iwf_management_file
{
    char name[30];//文件名称 '\0'结尾
    uint32_t offset;//偏移量
    uint32_t length;//长度
}iwf_management_file_t;
